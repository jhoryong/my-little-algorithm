```js
function isPrime(num){
    if (num === 1) // 1은 소수가 아니다.
      return false;

    if (num === 2) // 2는 소수이다.
      return true;

    if (num % 2 === 0) // 짝수는 소수가 아니다.
      return false;

    const sqrt = Math.floor(Math.sqrt(num)); // 제곱근

    for (let i = 3; i <= sqrt; i += 2){
      if (num % i === 0)
        return false;
    }
    return true;
  }
```

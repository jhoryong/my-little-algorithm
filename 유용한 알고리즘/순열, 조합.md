```js
function solution(M, arr) {
    let answer = [];
    let temp = Array(M).fill(0);
    let check = Array(arr.length).fill(0);
    function DFS(L) {
        if (L === M){
            answer.push(temp.slice());
        }
        else{
            for (let i = 0; i < arr.length; i++) {
                if (check[i]===0) {
                    temp[L] = arr[i];
                    check[i] = 1;
                    DFS(L+1);
                    check[i]=0;
                }
            }
        }
    }
    DFS(0);
    return answer;
}
```

```js
            answer.push(temp.slice());
        }
```

```js
function solution(N, M) {
    let answer = [];
    let temp = Array(M).fill(0);
    function DFS(L, S) {
        if (L === M) {
            answer.push(temp.slice())
        }
        else {
            for (let i=S; i<=N; i++){
                temp[L]=i;
                DFS(L+1, i+1);
            }
        }
    }
    DFS(0, 1)
    return answer;
}
```

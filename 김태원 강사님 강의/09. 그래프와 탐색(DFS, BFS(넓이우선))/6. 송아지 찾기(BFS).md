function solution(S, E) {
    debugger;
    let answer = 0;
    let check = Array(10001).fill(0);
    let dist = Array(10001).fill(0);
    let queue = [];
    check[S] = 1;
    queue.push(S);
    while (queue.length) {
        let X = queue.shift();
        for (let next of [X-1, X+1, X+5]) {
            if (next === E) return dist[X] + 1;
            if (next > 0 && next <= 10000 && check[next] === 0) {
                check[next] = 1;
                queue.push(next);
                dist[next] = dist[X] + 1;
            }
        }
    }
    return answer;
}

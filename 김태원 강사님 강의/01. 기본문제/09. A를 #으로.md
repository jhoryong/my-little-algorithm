### 문제
대문자로 이루어진 영어단어가 입력되면 단어에 포함된 ‘A'를 모두 ’#‘으로 바꾸어 출력하는 프로그램을 작성하세요.

### 내 풀이
```js
function solution(str) {
  let answer = '';
  for (let char of str) {
    if (char === 'A')
      answer += '#';
    else
      answer += char;
  }
  return answer;
}
```

### 정답
```js
// sol1
function solution(s){
  let answer="";
  for(let x of s){
      if(x=='A') answer+='#';
      else answer+=x;
  }
  return answer;
}

// sol2: 
function solution(s){
    let answer=s;
    answer=answer.replace(/A/g, "#");
    return answer;
}
```

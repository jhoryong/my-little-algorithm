### 내 풀이
```js
function solution(m, arr) {
  let p = 0, answer = 0;
  while (p < arr.length) {
    let sum = 0;
    for (let i = p; i < arr.length; i++){
      sum += arr[i];
      if (sum === m) {
        answer++;
        break;
      }
      else if (sum > m) break;
    }
    p++;
  }
  return answer;
}
```

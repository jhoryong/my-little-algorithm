### 내 풀이
```js
function solution(arr1, arr2) {
  let n = arr1.length, m = arr2.length, answer = [];
  let p1 = p2 = 0;
  while (p1 < n && p2 < m){
    if (arr1[p1] === arr2[p2]) {
      answer.push(arr1[p1]);
      p2++;
    }
    p1++;
  }
  return answer.sort((a, b)=>a - b);
}
```
